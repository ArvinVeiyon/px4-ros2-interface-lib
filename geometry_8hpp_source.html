58 {

60  Eigen::Matrix<Type, 3, 3> dcm = q.toRotationMatrix();
61 
62  angles.y() = asin(-dcm.coeff(2, 0));
63 
64  if ((std::fabs(angles.y() - static_cast<Type>(M_PI / 2))) < static_cast<Type>(1.0e-3)) {
65  angles.x() = 0;
66  angles.z() = atan2(dcm.coeff(1, 2), dcm.coeff(0, 2));
67  } else if ((std::fabs(angles.y() + static_cast<Type>(M_PI / 2))) < static_cast<Type>(1.0e-3)) {
68  angles.x() = 0;
69  angles.z() = atan2(-dcm.coeff(1, 2), -dcm.coeff(0, 2));
70  } else {
71  angles.x() = atan2(dcm.coeff(2, 1), dcm.coeff(2, 2));
72  angles.z() = atan2(dcm.coeff(1, 0), dcm.coeff(0, 0));
73  }
