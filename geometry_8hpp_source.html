20 Type wrapPi(Type angle)
21 {
22  while (angle >= M_PI) {
23  angle -= 2.0 * M_PI;
24  }

37 template<typename Type>
38 Eigen::Matrix<Type, 3, 1> quaternionToEulerRPY(const Eigen::Quaternion<Type> & q)
39 {
40  Eigen::Matrix<Type, 3, 1> angles;
41  Eigen::Matrix<Type, 3, 3> dcm = q.toRotationMatrix();
42 
43  angles.y() = asin(-dcm.coeff(2, 0));
44 
45  if ((std::fabs(angles.y() - static_cast<Type>(M_PI / 2))) < static_cast<Type>(1.0e-3)) {
46  angles.x() = 0;
47  angles.z() = atan2(dcm.coeff(1, 2), dcm.coeff(0, 2));
48  } else if ((std::fabs(angles.y() + static_cast<Type>(M_PI / 2))) < static_cast<Type>(1.0e-3)) {
49  angles.x() = 0;
50  angles.z() = atan2(-dcm.coeff(1, 2), -dcm.coeff(0, 2));
51  } else {
52  angles.x() = atan2(dcm.coeff(2, 1), dcm.coeff(2, 2));
53  angles.z() = atan2(dcm.coeff(1, 0), dcm.coeff(0, 0));
54  }
55 
56  return angles;
57 }

65 template<typename Type>
66 Type quaternionToRoll(const Eigen::Quaternion<Type> & q)
67 {
68  Type x = q.x();
69  Type y = q.y();
70  Type z = q.z();
71  Type w = q.w();
72 
73  return std::atan2(2.0 * (w * x + y * z), w * w - x * x - y * y + z * z);
74 }

87  Type z = q.z();
